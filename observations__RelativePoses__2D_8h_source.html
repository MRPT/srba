00058             {
00059                 const RelativePoses_2D::obs_data_t & kf0 = new_kf_obs[i], &kf1 = old_kf_obs[i];
00060                 if ( (kf0.x!=0 || kf0.y!=0 || kf0.yaw!=0) && 
00061                      (kf1.x!=0 || kf1.y!=0 || kf1.yaw!=0) )
00062                      continue; // skip.
00063                 const mrpt::poses::CPose2D new_obs(kf0.x,kf0.y,kf0.yaw);

00065                 pose_new_kf_wrt_old_kf = POSE(old_obs-new_obs);
00066                 return true;
00067             }
