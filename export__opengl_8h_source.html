00076                 //o->setPose(...);  // At the origin

00079

00103                         p2.z_incr( options.draw_kf_hierarchical_height );
00104                 }
00105                 gl_edges->appendLine(p1.x(),p1.y(),p1.z(), p2.x(),p2.y(),p2.z());

00136         children_depths[root_keyframe]=0;
